**Service 1: SearXNG**
```yaml
  # ==================== SearXNG (Search Engine) ====================
  searxng:
    <<: *security-defaults
    image: docker.io/searxng/searxng:${SEARXNG_VERSION:-2024.12.23-42c82f418}
    container_name: agmind-searxng
    restart: on-failure:5
    mem_limit: ${SEARXNG_MEM_LIMIT:-256m}
    profiles:
      - searxng
    environment:
      - SEARXNG_BASE_URL=http://localhost:8888/
    volumes:
      - ../templates/searxng-settings.yml:/etc/searxng/settings.yml:ro
    healthcheck:
      test: ["CMD-SHELL", "wget -qO- --timeout=5 http://localhost:8080/healthz || exit 1"]
      interval: ${HEALTHCHECK_INTERVAL:-30s}
      timeout: 10s
      retries: ${HEALTHCHECK_RETRIES:-5}
      start_period: 15s
    networks:
      - agmind-backend
```
Note: SearXNG listens on 8080 internally. We publish 8888 on host via `ports: ["${SEARXNG_PORT:-8888}:8080"]`.


Note: SearXNG does not publish semver releases; they use date+commit hash tags on DockerHub. `2024.12.23-42c82f418` is a known stable tag. Open Notebook uses `v1-latest` as their recommended production tag (no semver releases yet).

**templates/searxng-settings.yml** — create this file with the following content:

```yaml
# SearXNG settings for AGMind
# Ref: https://docs.searxng.org/admin/settings/settings.html

use_default_settings: true

general:
  instance_name: "AGMind Search"

search:
  safe_search: 0
  autocomplete: ""
  default_lang: "ru-RU"
  formats:
    - html
    - json

server:
  secret_key: "__SEARXNG_SECRET_KEY__"
  bind_address: "0.0.0.0"
  port: 8080
  limiter: false

engines:
  - name: google
    engine: google
    shortcut: g
    disabled: false
  - name: bing
    engine: bing
    shortcut: b
    disabled: false
  - name: duckduckgo
    engine: duckduckgo
    shortcut: ddg
    disabled: false
  - name: wikipedia
    engine: wikipedia
    shortcut: wp
    disabled: false
```

The `__SEARXNG_SECRET_KEY__` placeholder will be replaced by `lib/config.sh` during install (similar to how `__LITELLM_MASTER_KEY__` is handled). The `formats: [html, json]` line is critical — it enables the JSON API endpoint at `/search?q=test&format=json`.
