Modify `scripts/restore.sh` with three targeted fixes:

**Fix 1 — Standardize tmpdir to /opt/agmind/.restore_tmp:**

The current restore uses `mktemp -d "${data_dir}.old.XXXXXX"` for each volume restore (Qdrant line 191, Weaviate line 211, Dify storage line 234). The `mktemp` pattern creates directories adjacent to the data dir, which MAY be on a different mount than the backup source, causing cross-device move failures.

Replace all occurrences of the pattern:
```bash
temp_old=$(mktemp -d "${data_dir}.old.XXXXXX")
```

with a centralized tmpdir approach. At the top of the script (after `COMPOSE_FILE` and `BACKUP_BASE` definitions, around line 30), add:

```bash
RESTORE_TMP="${INSTALL_DIR}/.restore_tmp"
```

Then modify the `cleanup_restore()` function (lines 33-38) to ALSO clean up the tmpdir:
```bash
cleanup_restore() {
    if [[ "$SERVICES_DOWN" == "true" ]]; then
        echo -e "${YELLOW}Restore interrupted — restarting services...${NC}"
        cd "${INSTALL_DIR}/docker" && docker compose up -d 2>/dev/null || true
    fi
    # Clean up restore tmpdir
    if [[ -d "${RESTORE_TMP:-}" ]]; then
        rm -rf "$RESTORE_TMP"
    fi
}
trap cleanup_restore EXIT INT TERM
```

For each volume restore section (Qdrant, Weaviate, Dify storage), replace:
```bash
temp_old=$(mktemp -d "${data_dir}.old.XXXXXX")
```
with:
```bash
temp_old="${RESTORE_TMP}/$(basename "$data_dir").old"
mkdir -p "$temp_old"
```

Add `mkdir -p "$RESTORE_TMP"` once, right before the "Stop services" section (before line 135), so the tmpdir exists before any restore operation.

Also ensure the cleanup at end of each successful restore removes the specific temp_old, not with `rm -rf "${temp_old}"` (which it already does) — but also add a final cleanup after ALL restores succeed (before "Start all services" section, around line 310):
```bash
# Clean up restore tmpdir after successful restore
rm -rf "$RESTORE_TMP" 2>/dev/null || true
```

**Fix 2 — Explicit pipefail check on psql restore piping:**

The script already has `set -euo pipefail` on line 3, so pipe failures should propagate. However, the PostgreSQL restore commands on lines 163-164 use `if ! gunzip -c ... | docker compose ... exec -T db psql ...` which already captures the exit code correctly via the `if !` pattern.

Verify the existing pattern is correct (it is). No code change needed here — the `set -o pipefail` on line 3 plus the `if !` guard on the pipe already detects psql failures.

However, add a comment above the PostgreSQL restore section to document this:
```bash
    # set -o pipefail ensures gunzip|psql pipe failures are caught
```

**Fix 3 — Parser flag handling for CLI arguments:**

The current CLI parsing (lines 42-56) uses `realpath -m "$1"` which may not exist on all systems. Add a fallback:
```bash
if [[ -n "${1:-}" ]]; then
    RESTORE_DIR="$(realpath -m "$1" 2>/dev/null || readlink -f "$1" 2>/dev/null || echo "$1")"
```

Also, the script does not handle `--help` or invalid flags. Add before the RESTORE_DIR parsing (before line 42):
```bash
# Parse flags
while [[ $# -gt 0 ]]; do
    case "$1" in
        --auto-confirm) AUTO_CONFIRM=true; shift; continue ;;
        --help|-h)
            echo "Usage: restore.sh [BACKUP_DIR] [--auto-confirm]"
            echo "  BACKUP_DIR     Path to backup directory under ${BACKUP_BASE}"
            echo "  --auto-confirm Skip confirmation prompts"
            exit 0
            ;;
        --*) echo -e "${RED}Unknown option: $1${NC}"; exit 1 ;;
        *) break ;;
    esac
done
```

Then update the RESTORE_DIR block to use `"${1:-}"` (which it already does).

File structure:
```bash
#!/usr/bin/env bats

# test_backup.bats — Backup/Restore validation tests

setup() {
    export ROOT_DIR="$(cd "$(dirname "$BATS_TEST_FILENAME")/.." && pwd)"
    export INSTALL_DIR="$(mktemp -d)"
    mkdir -p "${INSTALL_DIR}/docker"
    mkdir -p "${INSTALL_DIR}/docker/volumes/app/storage"
    mkdir -p "${INSTALL_DIR}/scripts"
}

teardown() {
    rm -rf "$INSTALL_DIR"
}

# --- Syntax validation ---

@test "backup.sh passes bash -n syntax check" {
    run bash -n "${ROOT_DIR}/scripts/backup.sh"
    [ "$status" -eq 0 ]
}

@test "restore.sh passes bash -n syntax check" {
    run bash -n "${ROOT_DIR}/scripts/restore.sh"
    [ "$status" -eq 0 ]
}

# --- Restore tmpdir pattern ---

@test "restore.sh uses .restore_tmp not mktemp" {
    run grep "mktemp" "${ROOT_DIR}/scripts/restore.sh"
    [ "$status" -ne 0 ]
}

@test "restore.sh defines RESTORE_TMP variable" {
    run grep "RESTORE_TMP=" "${ROOT_DIR}/scripts/restore.sh"
    [ "$status" -eq 0 ]
}

@test "restore.sh cleans RESTORE_TMP in cleanup trap" {
    run grep -A5 "cleanup_restore" "${ROOT_DIR}/scripts/restore.sh"
    [ "$status" -eq 0 ]
    [[ "$output" == *"RESTORE_TMP"* ]]
}

# --- Pipefail ---

@test "restore.sh has set -euo pipefail" {
    run head -5 "${ROOT_DIR}/scripts/restore.sh"
    [[ "$output" == *"set -euo pipefail"* ]]
}

@test "backup.sh has set -euo pipefail" {
    run head -5 "${ROOT_DIR}/scripts/backup.sh"
    [[ "$output" == *"set -euo pipefail"* ]]
}

# --- Security ---

@test "backup.sh sets umask 077" {
    run grep "umask 077" "${ROOT_DIR}/scripts/backup.sh"
    [ "$status" -eq 0 ]
}

@test "backup.sh requires root" {
    run grep "EUID.*-ne 0\|id -u.*-ne 0" "${ROOT_DIR}/scripts/backup.sh"
    [ "$status" -eq 0 ]
}

@test "restore.sh requires root" {
    run grep "id -u.*-ne 0" "${ROOT_DIR}/scripts/restore.sh"
    [ "$status" -eq 0 ]
}

# --- Exclusive lock ---

@test "backup.sh uses exclusive lock" {
    run grep "flock" "${ROOT_DIR}/scripts/backup.sh"
    [ "$status" -eq 0 ]
}

@test "restore.sh uses exclusive lock" {
    run grep "flock" "${ROOT_DIR}/scripts/restore.sh"
    [ "$status" -eq 0 ]
}

# --- Parser and flags ---

@test "restore.sh supports --auto-confirm flag" {
    run grep "\-\-auto-confirm" "${ROOT_DIR}/scripts/restore.sh"
    [ "$status" -eq 0 ]
}

@test "restore.sh supports --help flag" {
    run grep "\-\-help" "${ROOT_DIR}/scripts/restore.sh"
    [ "$status" -eq 0 ]
}

@test "restore.sh validates INSTALL_DIR path" {
    run grep 'INSTALL_DIR.*==.*/opt/agmind' "${ROOT_DIR}/scripts/restore.sh"
    [ "$status" -eq 0 ]
}

# --- Backup output structure ---

@test "backup.sh creates checksums file" {
    run grep "sha256sum" "${ROOT_DIR}/scripts/backup.sh"
    [ "$status" -eq 0 ]
}

@test "backup.sh has cleanup trap on failure" {
    run grep "cleanup_backup" "${ROOT_DIR}/scripts/backup.sh"
    [ "$status" -eq 0 ]
}

@test "backup.sh supports retention by count" {
    run grep "BACKUP_RETENTION_COUNT" "${ROOT_DIR}/scripts/backup.sh"
    [ "$status" -eq 0 ]
}

# --- Restore does not copy .age keys ---

@test "restore.sh does not blindly copy .age keys from backup" {
    # The restore should use age keys from INSTALL_DIR, not copy from backup source
    # The age_keys restore is gated behind user confirmation (RESTORE_CONFIG)
    run grep -c "cp.*age_keys" "${ROOT_DIR}/scripts/restore.sh"
    # Should only appear in the optional config restore block (user-confirmed)
    [ "$status" -eq 0 ]
}
```
