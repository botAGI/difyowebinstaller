Extend .github/workflows/sync-release.yml to also regenerate release-manifest.json.

The existing generate-manifest.sh requires Docker for digest resolution (docker manifest inspect), which won't work in a basic CI runner without authentication. Instead, use a lightweight approach: update the manifest JSON inline using python3 (available on ubuntu-latest) by reading versions.env and patching the tag fields in release-manifest.json. Digests can remain empty (they are optional and not used by the installer at runtime).

In the "Update repo files" step, AFTER copying versions.env and writing RELEASE, add manifest regeneration:

```bash
          # Update release-manifest.json from versions.env
          TAG="${{ github.event.release.tag_name }}"
          RELEASE_DATE="${{ github.event.release.published_at }}"

          python3 -c "
import json, re, sys
from datetime import datetime

# Parse versions.env into dict
versions = {}
with open('templates/versions.env') as f:
    for line in f:
        line = line.strip()
        if not line or line.startswith('#'):
            continue
        if '=' in line:
            k, v = line.split('=', 1)
            versions[k.strip()] = v.strip()

# Load existing manifest
with open('templates/release-manifest.json') as f:
    manifest = json.load(f)

# Update top-level fields
manifest['version'] = sys.argv[1]
manifest['release_date'] = sys.argv[2][:10] if len(sys.argv) > 2 and sys.argv[2] else datetime.utcnow().strftime('%Y-%m-%d')

# Version key -> manifest image key mapping
key_map = {
    'DIFY_VERSION': ['dify-api', 'dify-web'],
    'OPENWEBUI_VERSION': ['open-webui'],
    'OLLAMA_VERSION': ['ollama'],
    'POSTGRES_VERSION': ['postgres'],
    'REDIS_VERSION': ['redis'],
    'WEAVIATE_VERSION': ['weaviate'],
    'QDRANT_VERSION': ['qdrant'],
    'SANDBOX_VERSION': ['dify-sandbox'],
    'SQUID_VERSION': ['squid'],
    'NGINX_VERSION': ['nginx'],
    'CERTBOT_VERSION': ['certbot'],
    'PLUGIN_DAEMON_VERSION': ['dify-plugin-daemon'],
    'DOCLING_SERVE_VERSION': ['docling-serve'],
    'AUTHELIA_VERSION': ['authelia'],
    'GRAFANA_VERSION': ['grafana'],
    'PORTAINER_VERSION': ['portainer'],
    'NODE_EXPORTER_VERSION': ['node-exporter'],
    'CADVISOR_VERSION': ['cadvisor'],
    'PROMETHEUS_VERSION': ['prometheus'],
    'ALERTMANAGER_VERSION': ['alertmanager'],
    'LOKI_VERSION': ['loki'],

}

images = manifest.get('images', {})
for env_key, image_keys in key_map.items():

        for img_key in image_keys:
            if img_key in images:

with open('templates/release-manifest.json', 'w') as f:

    f.write('\n')

print(f'Updated manifest: version={manifest[\"version\"]}, date={manifest[\"release_date\"]}, images={len(images)}')
" "$TAG" "${RELEASE_DATE:-}"
```

The full updated workflow file should be:

```yaml
name: Sync Release to Repo

on:
  release:
    types: [published]

permissions:
  contents: write

jobs:
  sync:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v5
        with:
          ref: main
          token: ${{ secrets.GITHUB_TOKEN }}

      - name: Download versions.env from release
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        run: |
          TAG="${{ github.event.release.tag_name }}"
          echo "Release tag: ${TAG}"

          # Download versions.env asset
          gh release download "$TAG" --pattern "versions.env" --output /tmp/versions.env

          if [[ ! -s /tmp/versions.env ]]; then
            echo "::error::versions.env asset not found in release ${TAG}"
            exit 1
          fi

          echo "Downloaded versions.env ($(wc -l < /tmp/versions.env) lines)"

      - name: Update repo files
        run: |
          TAG="${{ github.event.release.tag_name }}"
          RELEASE_DATE="${{ github.event.release.published_at }}"

          cp /tmp/versions.env templates/versions.env
          echo "$TAG" > RELEASE

          # Regenerate release-manifest.json from updated versions.env
          python3 -c "
          import json, sys
          from datetime import datetime

          versions = {}
          with open('templates/versions.env') as f:
              for line in f:
                  line = line.strip()
                  if not line or line.startswith('#'):
                      continue
                  if '=' in line:
                      k, v = line.split('=', 1)
                      versions[k.strip()] = v.strip()

          with open('templates/release-manifest.json') as f:
              manifest = json.load(f)

          manifest['version'] = sys.argv[1]
          rd = sys.argv[2] if len(sys.argv) > 2 and sys.argv[2] else ''
          manifest['release_date'] = rd[:10] if rd else datetime.utcnow().strftime('%Y-%m-%d')

          key_map = {
              'DIFY_VERSION': ['dify-api', 'dify-web'],
              'OPENWEBUI_VERSION': ['open-webui'],
              'OLLAMA_VERSION': ['ollama'],
              'POSTGRES_VERSION': ['postgres'],
              'REDIS_VERSION': ['redis'],
              'WEAVIATE_VERSION': ['weaviate'],
              'QDRANT_VERSION': ['qdrant'],
              'SANDBOX_VERSION': ['dify-sandbox'],
              'SQUID_VERSION': ['squid'],
              'NGINX_VERSION': ['nginx'],
              'CERTBOT_VERSION': ['certbot'],
              'PLUGIN_DAEMON_VERSION': ['dify-plugin-daemon'],
              'DOCLING_SERVE_VERSION': ['docling-serve'],
              'AUTHELIA_VERSION': ['authelia'],
              'GRAFANA_VERSION': ['grafana'],
              'PORTAINER_VERSION': ['portainer'],
              'NODE_EXPORTER_VERSION': ['node-exporter'],
              'CADVISOR_VERSION': ['cadvisor'],
              'PROMETHEUS_VERSION': ['prometheus'],
              'ALERTMANAGER_VERSION': ['alertmanager'],
              'LOKI_VERSION': ['loki'],
              'PROMTAIL_VERSION': ['promtail'],
              'VLLM_VERSION': ['vllm'],
              'TEI_VERSION': ['tei'],
              'PIPELINES_VERSION': ['pipelines'],
          }

          images = manifest.get('images', {})
          for env_key, image_keys in key_map.items():
              if env_key in versions:
                  for img_key in image_keys:
                      if img_key in images:
                          images[img_key]['tag'] = versions[env_key]

          manifest['images'] = images

          with open('templates/release-manifest.json', 'w') as f:
              json.dump(manifest, f, indent=2)
              f.write('\n')

          print(f'Manifest updated: version={manifest[\"version\"]}, date={manifest[\"release_date\"]}, {len(images)} images')
          " "$TAG" "${RELEASE_DATE:-}"

          if git diff --quiet; then
            echo "No changes -- repo already matches release ${TAG}"
            exit 0
          fi

          echo "changes=true" >> "$GITHUB_ENV"
          echo "--- Diff ---"
          git diff

      - name: Commit and push
        if: env.changes == 'true'
        run: |
          TAG="${{ github.event.release.tag_name }}"
          git config user.name "github-actions[bot]"
          git config user.email "41898282+github-actions[bot]@users.noreply.github.com"
          git add templates/versions.env templates/release-manifest.json RELEASE
          git commit -m "chore(release): sync ${TAG} versions.env, manifest and RELEASE"
          git push
```
